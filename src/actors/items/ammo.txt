	}

ACTOR ArgBFGAmmoSpawner
{
	Scale 0.66
	+FORCEYBILLBOARD
	States
	{
	Spawn:
		BFGC A 0 NoDelay A_SpawnItemEx("ArgBFGAmmo",0,0,0,0,0,0,0,SXF_ISTRACER|SXF_TRANSFERSPECIAL)
	Idle:
		BFGC A 1
		BFGC A 1 {
			if(CheckClass("ArgBFGAmmo", AAPTR_TRACER)) {
				return state(0);
			}
			return state("Vanish");
		}
		Wait
	Vanish:
		TNT1 A 0
		Stop
	}
}
